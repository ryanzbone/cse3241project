CREATE VIEW CustomerTotalSpent AS
SELECT N.id, N.fname, N.lname, Sum(M.amount)
  FROM Names N, Customer C, MonetaryTransaction M
 WHERE N.id = C.eid
   AND M.epayer = C.eid
   GROUP BY N.id;
CREATE VIEW quantitysoldperbook
AS
  SELECT O.bid,
         Sum(O.quantity) AS numberSold
  FROM   ordercontents O
  GROUP  BY O.bid;
CREATE VIEW AuthorProfitQuantity AS
SELECT A.fname, A.mname, A.lname, Sum(P.profit), Sum(T.numberSold)
  from Names A,
 (SELECT O.bid AS isbn, Sum(O.quantity) AS numberSold
  FROM OrderContents O
  GROUP BY O.bid) T,
(SELECT T.isbn, T.Sum * B.price AS profit
From
(SELECT O.bid AS isbn, Sum(O.quantity) AS Sum
FROM OrderContents O
GROUP BY O.bid) T,
Book B
Where B.isbn = T.isbn
) P,
BookAuthor BA
where t.isbn = p.isbn
and ba.isbn = t.isbn
and BA.nameid = A.id
group by ba.nameid;
COMMIT;
PRAGMA foreign_keys=ON;
